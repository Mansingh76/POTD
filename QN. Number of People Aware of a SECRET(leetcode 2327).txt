2327. Number of People Aware of a Secret



class Solution {
    public int peopleAwareOfSecret(int n, int delay, int forget) {
        int mod = 1_000_000_007;
        long[] dp = new long[n + 1];  // dp[i] = number of people who learn on day i
        dp[1] = 1; 
        
        long share = 0;  // running sum of people currently eligible to share
        
        for (int day = 2; day <= n; day++) {
            // Add people who become eligible today
            if (day - delay >= 1) {
                share = (share + dp[day - delay]) % mod;
            }
            // Remove people who forget today
            if (day - forget >= 1) {
                share = (share - dp[day - forget] + mod) % mod;
            }
            dp[day] = share;
        }
        
        // Count those who still remember on day n
        long ans = 0;
        for (int i = n - forget + 1; i <= n; i++) {
            if (i >= 1) ans = (ans + dp[i]) % mod;
        }
        return (int) ans;
    }
}
