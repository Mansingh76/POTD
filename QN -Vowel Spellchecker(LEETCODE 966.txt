QN -Vowel Spellchecker(LEETCODE 966)


class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Set<String> exactWords = new HashSet<>();
        Map<String, String> caseInsensitiveMap = new HashMap<>();
        Map<String, String> vowelInsensitiveMap = new HashMap<>();
        
        for (String word : wordlist) {
            exactWords.add(word);
            
            String lower = word.toLowerCase();
            caseInsensitiveMap.putIfAbsent(lower, word);
            
            String devoweled = devowel(lower);
            vowelInsensitiveMap.putIfAbsent(devoweled, word);
        }
        
        String[] result = new String[queries.length];
        for (int i = 0; i < queries.length; i++) {
            String q = queries[i];
            if (exactWords.contains(q)) {
                result[i] = q;
            } else {
                String lower = q.toLowerCase();
                if (caseInsensitiveMap.containsKey(lower)) {
                    result[i] = caseInsensitiveMap.get(lower);
                } else {
                    String devoweled = devowel(lower);
                    result[i] = vowelInsensitiveMap.getOrDefault(devoweled, "");
                }
            }
        }
        return result;
    }
    
    private String devowel(String word) {
        return word.replaceAll("[aeiou]", "*");
    }
}
